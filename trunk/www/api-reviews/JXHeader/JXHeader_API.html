package org.jdesktop.swingx;

import java.awt.Color;
import java.awt.Font;
import java.awt.GridBagConstraints;
import java.awt.GridBagLayout;
import java.awt.Insets;
import javax.swing.Icon;
import javax.swing.JLabel;
import javax.swing.JSeparator;
import javax.swing.UIManager;
import org.jdesktop.swingx.painter.gradient.BasicGradientPainter;
import org.jdesktop.swingx.plaf.HeaderUI;
import org.jdesktop.swingx.plaf.JXHeaderAddon;
import org.jdesktop.swingx.plaf.LookAndFeelAddons;

/**
 * <p>A simple component that consists of a title, a description, and an icon.
 * An example of such a component can be seen on
 * <a href="http://jext.free.fr/header.png">Romain Guys ProgX website</a></p>
 *
 * <p>The <code>JXHeader</code> is sufficiently configurable to be useable. The 
 * title and description area accept HTML (the same set of HTML that 
 * {@link org.jdesktop.swingx.JXEditorPane} supports), and thus allows a great 
 * deal of flexibility. The icon and the various bits of text are configurable. 
 *
 * <p>Here is some example code:<br/>
 * <pre><code>
 *      JXHeader header = new JXHeader();
 *      header.setTitle("Timing Framework Spline Editor");
 *      header.setDescription("&lt;html&gt;&lt;body&gt;" +
 *          "Drag control points in the display to change the shape of the spline&lt;br/&gt;" +
 *          "Click the Copy Code button to generate the corrosponding Java code." +
 *          "&lt;/body&gt;&lt;/html&gt;");
 *      Icon icon = new ImageIcon(getClass().getResource("tools.png"));
 *      header.setIcon(icon);
 * </code></pre></p>
 * 
 * @author rbair
 */
public class JXHeader extends JXPanel {
    /**
     * @see #getUIClassID // *
     * @see #readObject
     */
    public static final String uiClassID = "HeaderUI";
    
    /** 
     * Creates a new instance of JXHeader
     */
    public JXHeader();
    
    /**
     * Returns the look and feel (L&F) object that renders this component.
     * 
     * @return the HeaderUI object that renders this component
     */
    @Override
    public HeaderUI getUI();

    /**
     * Sets the look and feel (L&F) object that renders this component.
     * 
     * @param ui
     *            the HeaderUI L&F object
     * @see javax.swing.UIDefaults#getUI
     * @beaninfo bound: true hidden: true attribute: visualUpdate true
     *           description: The UI object that implements the Component's
     *           LookAndFeel.
     */
    public void setUI(HeaderUI ui);

    /**
     * Returns a string that specifies the name of the L&F class that renders
     * this component.
     * 
     * @return "HeaderUI"
     * @see javax.swing.JComponent#getUIClassID
     * @see javax.swing.UIDefaults#getUI
     * @beaninfo expert: true description: A string that specifies the name of
     *           the L&F class.
     */
    @Override
    public String getUIClassID();

    /**
     * Notification from the <code>UIManager</code> that the L&F has changed.
     * Replaces the current UI object with the latest version from the
     * <code>UIManager</code>.
     * 
     * @see javax.swing.JComponent#updateUI
     */
    @Override
    public void updateUI();
    
    /**
     * @inheritDoc
     * Passes enabled state on to the child components
     */
    @Override
    public void setEnabled(boolean enabled);
    
    /**
     * Sets the title of the <code>JXHeader</code>.
     * 
     * @param title can be any value, including HTML 3.2.
     */
    public void setTitle(String title);
    
    /**
     * Gets the title.
     * 
     * @return the title
     */
    public String getTitle();
    
    /**
     * Sets the description.
     * 
     * @param description any value. May be HTML based on HTML 3.2
     */
    public void setDescription(String description);
    
    /**
     * Gets the description.
     * 
     * @return description
     */
    public String getDescription();
    
    /**
     * Sets the icon to use with the <code>JXHeader</code>.
     * 
     * @param icon the icon to use. May be null.
     */
    public void setIcon(Icon icon);
    
    /**
     * Gets the icon.
     * 
     * @return icon
     */
    public Icon getIcon();
}
